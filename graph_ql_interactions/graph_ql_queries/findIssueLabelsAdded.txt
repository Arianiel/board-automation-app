query findIssueLabelsAdded {
  node(id: "<ISSUE>") {
    ... on Issue {
      number
      id
      labels(first: 10){
        nodes {
            name
        }
      }
      timelineItems(last: 20, itemTypes: [LABELED_EVENT]){
        totalCount
        edges {
            node {
                __typename
                ... on LabeledEvent{
                    createdAt
                    label {
                        name
                    }
                }
            }
        }
      }
    }
  }
}
