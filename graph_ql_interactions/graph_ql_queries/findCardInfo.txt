query findCardInfo {
    organization(login: "<ORG_NAME>"){
        projectV2(number: <PROJ_NUM>){
            ... on ProjectV2 {
                items(first: 100, after: <AFTER>){
                    nodes {
                        id
                        ... on ProjectV2Item{
                            type
                            content {
                                ... on Issue {
                                    id
                                    number
                                    labels (first: 20){
                                        nodes {
                                            name
                                        }
                                    }
                                    repository {
                                        name
                                    }
                                }
                                ... on DraftIssue {
                                    title
                                    id
                                }
                            }
                            fieldValues(first: 20) {
                                nodes {
                                    ... on ProjectV2ItemFieldSingleSelectValue {
                                        name
                                        field {
                                            ... on ProjectV2SingleSelectField {
                                                name
                                            }
                                        }
                                    }
                                    ... on ProjectV2ItemFieldNumberValue {
                                        number
                                        field {
                                            ... on ProjectV2Field {
                                                name
                                            }
                                        }
                                    }
                                }
                            }
                        }
                    }
                    pageInfo {
                        endCursor
                        startCursor
                        hasNextPage
                        hasPreviousPage
                    }
                }
            }
        }
    }
}
